function temporarySwap(input){
    var left=null;
    var right=null;
    var length = input.length;
    for(left=0,right=length-1;left<right;left+=1,right-=1){
        var temporary = input[left];
        input[left]=input[right];
        input[right]=temporary;
    }
    return input;
}

            function splittest(line){
       //var line2 = line.replace(/[/g, "");
        var arr = [];
        var arrytest = line.split("],");
        for (var i = 0; i < arrytest.length; i++) {
            var arr2 = [];
            var replaced = arrytest[i].replace("[","");
            replaced =  replaced.replace("[","");
            var secondvar = replaced.split(",");
            for(var j = 0;j<secondvar.length;j++){
                var replaced2 = secondvar[j].replace("]","");
                replaced2 = replaced2.replace("]","");
                replaced2 = replaced2.replace("'","");
                replaced2 = replaced2.replace("'","");
                arr2.push(replaced2);
            }
            arr.push(arr2);

           }
          //alert(arr);
          return arr;
    }
